alter table public.bookings
  add column if not exists service_id uuid null,
  add column if not exists group_size integer default 1,
  add column if not exists total_price numeric null,
  add column if not exists assigned_to text null;

-- Recommended indexes for date/provider lookups
create index if not exists bookings_provider_date_idx on public.bookings(provider_id, date);
